import random


def rola_dados(dado, qtde):
    """
    Rola um determinado número de dados e os imprime lado a lado.

    Argumentos:
    dado (int) - O tamanho do dado.
    qtde (int) - O número de dados a serem rolados.

    Retorna: None
    """
    if qtde > 0:
        mensagem = ""
        for rolagem in range(1, qtde + 1):
            resultado = random.randint(1, dado)
            print("Lançamento n.", rolagem, "-", resultado)
            mensagem += str(resultado) + " "

        print()
        print(qtde, "dado(s) de", dado, "lados:")
        print(mensagem)


def valida_dados(info):
    """
    Retorna True se um determinado dado é numérico maior que zero.
    Imprime na tela caso seja um dado inválido.

    Argumento:
    info (str) - A informação que precisa ser validada.

    Retorna: bool - True se o dado for numérico, False caso contrário
    """
    if not info.isnumeric():
        print("A informação passada não é válida!")
        return False

    if int(info) == 0:
        print("O número passado deve ser maior que zero!")
        return False

    return True


def dados():
    """Função principal, contém a lógica principal do programa."""
    while True:
        dado = input("Forneça o tamanho do dado que será rolado (ENTER para sair): ")
        if dado == "":
            break
        if valida_dados(dado):
            while True:
                qtde = input("Forneça o número de dados que serão rolados (em branco == 1): ")
                if qtde == "":
                    rola_dados(int(dado), 1)
                    break

                if valida_dados(qtde):
                    rola_dados(int(dado), int(qtde))
                    break


def calcula_operacao(num1, num2, oper):
    if oper == "+":
        return num1 + num2
    elif oper == "-":
        return num1 - num2
    elif oper == "*":
        return num1 * num2

    return num1 / num2

def calculadora():
    resultado = 0.0
    operacao = "X"
    while True:
        print(resultado)

        if not resultado and operacao == "X":
            operando = input("Informe um número, ou aperte ENTER para sair: ")

            if not operando:
                break

            resultado = float(operando)

        operacao = input("Operação (+, -, *, /), X para zerar ou ENTER para sair: ")

        if not operacao:
            break

        if operacao == "X":
            resultado = 0
            continue

        operando = input("Informe um número, ou aperte ENTER para sair: ")

        if not operando:
            break

        resultado = calcula_operacao(resultado, float(operando), operacao)

calculadora()
